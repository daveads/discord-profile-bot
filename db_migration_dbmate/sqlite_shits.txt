CREATE TABLE IF NOT EXISTS user(

		    id INTEGER PRIMARY KEY,
            username VARCHAR(255),
            username_id VARCHAR(25),
            name VARCHAR(20),
            location VARCHAR(15),
            looking_for VARCHAR(30),
            hobbies VARCHAR(100),
            biography VARCHAR(150),
			age INTEGER(10),
            profile_date VARCHAR(12)
            )
);
